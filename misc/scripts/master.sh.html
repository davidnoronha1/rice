 3 # Setup Colemak & Picom
 4 case $1 in
 5     launch)
 6         printf "
 7 \e[1mLAUNCH TASK:\e[0m
 8 - Killing Picom\n"
 9         pkill -e picom
10         printf "\n- Setting Colemak Keyboard Layout\n"
11         xmodmap $( dirname $0 )/xmodmap.colemak && xset r 66
12         printf "\n- Launching Picom\n\n"
13         (setsid picom -c -C 2> /dev/null &) # Fork so the shell can die.
14         exit
15         ;;
16 esac

82 # Compress File / Folder
83 compress () {
84     tar_f=$(mktemp)
85     # Tar if is directory
86     if [ -d "$1" ]; then
87         tar -cf "$tar_f" "$1"
88     else
89         tar_f=$1
90     fi
91     # Maximum compression
92     zstd -22 --ultra -T0 -z "$tar_f" -o "$(basename "$1").tar.zst"
93     rm $tar_f
94 }
